struct PX_DEPRECATED PxParticleCloth
{
    PxU32   startVertexIndex;
    PxU32   numVertices;
    PxReal  clothBlendScale;
    PxReal  restVolume;
    PxReal  pressure;
    PxU32   startTriangleIndex;
    PxU32   numTriangles;

    bool operator <= (const PxParticleCloth& other) const { return startVertexIndex <= other.startVertexIndex; }
    bool operator >= (const PxParticleCloth& other) const { return startVertexIndex >= other.startVertexIndex; }
    bool operator < (const PxParticleCloth& other) const { return startVertexIndex < other.startVertexIndex; }
    bool operator > (const PxParticleCloth& other) const { return startVertexIndex > other.startVertexIndex; }
    bool operator == (const PxParticleCloth& other) const { return startVertexIndex == other.startVertexIndex; }
};
